    }
};

        nil -> r |

    }

    match l {
        nil -> 0
        |
        h::t -> op (h) (reduce (t) (op))
    }
};

{
    if (n==0) {
        nil

    match l {
        nil -> 0 |
        x::t -> x + sum (t)
    }

    match l {
        nil -> 0 |

// Tests with lazy lists
// Prints first 100 numbers tripled
let intsfm = fn n => {
    n:? (intsfm (n+1))
};
let map = fn l, g => {
    match l {
        nil -> nil
    |
        h::t -> (g(h)):?(map(t)(g))
    }
};
let pfst = fn l,n => {
    if (n==0) {
        println(n)
    } else {
        match l {
            nil -> 0
            |
            h::t -> println(h); pfst (t) (n-1)
        }
    }
};
let l = intsfm (0);
let triple = fn n => { n*3 };
let m = map (l) (triple);
pfst (m) (100);;
